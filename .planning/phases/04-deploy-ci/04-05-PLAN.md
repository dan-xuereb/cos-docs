After Task 1's rollout is green, run the 6 smoke-test assertions against `http://10.70.0.102:30081/`. Each must return the expected substring; any failure aborts this task and routes back to diagnosis.

```bash
set -eo pipefail
BASE=http://10.70.0.102:30081

# Assertion 1: /health liveness endpoint
curl -fsS -o - -w "\nHTTP %{http_code}\n" "$BASE/health" | tee /tmp/a1.out
grep -q "healthy" /tmp/a1.out

# Assertion 2: landing page at / returns 200 with 'Xuer Labs'
curl -fsS -o - -w "\nHTTP %{http_code}\n" "$BASE/" | tee /tmp/a2.out
grep -q "Xuer Labs" /tmp/a2.out

# Assertion 3: directory-URL serving — /architecture/ returns 200 with Mermaid markup
curl -fsS -o - -w "\nHTTP %{http_code}\n" "$BASE/architecture/" | tee /tmp/a3.out
grep -q "Workspace Architecture" /tmp/a3.out
# Mermaid rendering marker (RESEARCH §5 + 03-03 evidence)
grep -qE 'class="mermaid"|<pre class="mermaid"' /tmp/a3.out

# Assertion 4: a random Python repo's pre-rendered API page has the passthrough marker
curl -fsS -o - -w "\nHTTP %{http_code}\n" "$BASE/COS-Core/api/" | tee /tmp/a4.out
grep -q "cos-docs-prerendered-api" /tmp/a4.out

# Assertion 5: /site-manifest.json is served at root with >=25 repos
curl -fsS "$BASE/site-manifest.json" | tee /tmp/a5.out | jq -e '.repos | length >= 25' >/dev/null
jq -e 'has("cos_docs_sha") and has("generated_at")' /tmp/a5.out >/dev/null

# Assertion 6: 404 page is the themed Material 404 (serves via try_files fallback to /404.html)
curl -sS -o /tmp/a6.out -w "%{http_code}\n" "$BASE/definitely-not-a-real-page/" > /tmp/a6.code
grep -qx "404" /tmp/a6.code
# Themed 404 contains Material's search/header markup; weak assertion — just check it is HTML not plaintext.
grep -q "<!DOCTYPE html" /tmp/a6.out

echo "ALL 6 ASSERTIONS PASSED"
```
